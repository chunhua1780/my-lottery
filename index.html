import sys import sqlite3 import pandas as pd from PyQt6.QtWidgets import QApplication, QMainWindow, QTableWidget, QTableWidgetItem, QVBoxLayout, QWidget, QLabel from PyQt6.QtCore import QTimer, Qt class EmpireCommand(QMainWindow): def __init__(self): super().__init__() self.setWindowTitle("帝国指挥中心 - 统计监测系统") self.setFixedSize(600, 400) self.init_ui() # 核心：设置定时器，每2秒刷一次数据库 self.timer = QTimer() self.timer.timeout.connect(self.load_data) self.timer.start(2000) def init_ui(self): layout = QVBoxLayout() self.label = QLabel("📊 实时刮奖中奖统计 (数据源: Local DB)") self.label.setStyleSheet("font-weight: bold; font-size: 16px; margin: 10px;") self.table = QTableWidget() self.table.setColumnCount(3) self.table.setHorizontalHeaderLabels(["玩家名称", "中奖金额", "流水单号"]) layout.addWidget(self.label) layout.addWidget(self.table) central_widget = QWidget() central_widget.setLayout(layout) self.setCentralWidget(central_widget) def load_data(self): try: conn = sqlite3.connect('data/data.db') df = pd.read_sql_query("SELECT user_name, prize_amount, sn FROM lottery_records ORDER BY id DESC LIMIT 50", conn) conn.close() self.table.setRowCount(len(df)) for i, row in df.iterrows(): self.table.setItem(i, 0, QTableWidgetItem(str(row['user_name']))) self.table.setItem(i, 1, QTableWidgetItem(f"${row['prize_amount']}")) self.table.setItem(i, 2, QTableWidgetItem(str(row['sn']))) except Exception as e: print(f"扫描数据库中... {e}") if __name__ == "__main__": app = QApplication(sys.argv) window = EmpireCommand() window.show() sys.exit(app.exec())
